\begin{align}
    \label{eqn:bin_count}
    N       &= \sum 1_t \\
    \label{eqn:bin_sum}
    Sum     &= \sum v_t \\
    \label{eqn:bin_sqsum}
    SqSum   &= \sum {v_t}^2 \\
    \label{eqn:bin_avg}
    \avg    &= Sum/N \\
    \label{eqn:bin_stddev}
    \stddev  &= SqSum - 2N\avg + N\avg^2
\end{align}

\begin{align}
    \label{eqn:bin_count2}
    N       &= \sum 1_t, \\
    \label{eqn:bin_sum2}
    Sum     &= \sum v_t, \\
    \label{eqn:bin_sqsum2}
    SqSum   &= \sum {v_t}^2, \\
    \label{eqn:bin_avg2}
    \avg    &= Sum/N, \\
    \label{eqn:bin_stddev2}
    \stddev  &= SqSum - 2N\avg + N\avg^2.
\end{align}
